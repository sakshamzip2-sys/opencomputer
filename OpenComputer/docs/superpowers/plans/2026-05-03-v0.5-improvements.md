```python
# opencomputer/cron/policy_digest.py
"""Daily digest cron — single Telegram with N pending policy_changes."""
import logging
import time
from datetime import datetime

logger = logging.getLogger(__name__)
_LAST_DIGEST_KEY = "_policy_digest_last_run"


def run_policy_digest(*, db, flags, send_fn=None) -> int:
    """Once per local-day, collect pending_approval rows and emit one digest.

    Returns count of recommendations included in the digest, or 0 if not
    yet time to fire today.
    """
    if not flags.read("policy_engine.digest_mode", True):
        return 0

    today = datetime.now().strftime("%Y-%m-%d")
    last_run_day = flags.read(_LAST_DIGEST_KEY, "")
    if last_run_day == today:
        return 0  # already fired today

    target_hour = int(flags.read("policy_engine.digest_hour_local", 9))
    if datetime.now().hour < target_hour:
        return 0  # too early

    with db._connect() as conn:
        rows = conn.execute(
            "SELECT id, knob_kind, target_id, reason, recommendation_engine_version "
            "FROM policy_changes "
            "WHERE status = 'pending_approval' "
            "ORDER BY ts_drafted DESC"
        ).fetchall()

    if not rows:
        flags.write(_LAST_DIGEST_KEY, today)
        return 0

    lines = [f"🤖 Policy engine digest — {len(rows)} pending recommendations:\n"]
    for r in rows:
        lines.append(
            f"  • {r['id'][:8]} {r['knob_kind']} → {r['target_id']}\n"
            f"    engine: {r['recommendation_engine_version']}\n"
            f"    reason: {r['reason']}\n"
            f"    approve: /policy-approve {r['id'][:8]}\n"
        )
    lines.append(
        "Use /policy-changes for full audit. Drafts auto-discard after 7 days."
    )
    digest_text = "\n".join(lines)

    if send_fn:
        try:
            import asyncio
            asyncio.run(send_fn(digest_text))
        except Exception as e:
            logger.warning("digest send failed: %s", e)

    flags.write(_LAST_DIGEST_KEY, today)
    return len(rows)
```

```python
# opencomputer/agent/policy_audit_log.py
"""Append-only HMAC chain over status transitions."""
import hashlib, hmac, sqlite3, time


class PolicyAuditLog:
    GENESIS = "0" * 64

    def __init__(self, conn, hmac_key):
        self._conn = conn
        self._key = hmac_key

    def append_transition(self, *, change_id, status, actor=None, reason=None):
        ts = time.time()
        prev = self._last_hmac()
        body = f"{prev}|{change_id}|{ts}|{status}|{actor or ''}|{reason or ''}"
        row_hmac = hmac.new(self._key, body.encode(), hashlib.sha256).hexdigest()
        self._conn.execute(
            "INSERT INTO policy_audit_log "
            "(change_id, ts, status, actor, reason, hmac_prev, hmac_self) "
            "VALUES (?, ?, ?, ?, ?, ?, ?)",
            (change_id, ts, status, actor, reason, prev, row_hmac),
        )
        self._conn.commit()

    def _last_hmac(self):
        row = self._conn.execute(
            "SELECT hmac_self FROM policy_audit_log ORDER BY id DESC LIMIT 1"
        ).fetchone()
        return row[0] if row else self.GENESIS

    def verify_chain(self):
        prev = self.GENESIS
        for row in self._conn.execute(
            "SELECT change_id, ts, status, actor, reason, hmac_prev, hmac_self "
            "FROM policy_audit_log ORDER BY id"
        ):
            cid, ts, status, actor, reason, hp, hs = row
            if hp != prev:
                return False
            body = f"{prev}|{cid}|{ts}|{status}|{actor or ''}|{reason or ''}"
            expected = hmac.new(self._key, body.encode(), hashlib.sha256).hexdigest()
            if expected != hs:
                return False
            prev = hs
        return True
```

```python
# opencomputer/evolution/engine_metrics.py
"""Read-only engine quality computation."""
from dataclasses import dataclass
import time


@dataclass(frozen=True, slots=True)
class EngineQuality:
    engine_version: str
    n_recommendations: int
    n_active: int
    n_expired_decayed: int
    n_reverted: int
    n_pending: int
    unrevert_rate: float          # (active + expired_decayed) / (recommendations - pending)
    revert_rate: float            # reverted / (recommendations - pending)
    days_window: int


def compute_engine_quality(db, engine_version=None, days=30):
    cutoff = time.time() - days * 86400
    with db._connect() as conn:
        params = [cutoff]
        sql = (
            "SELECT recommendation_engine_version AS engine, status, COUNT(*) AS n "
            "FROM policy_changes WHERE ts_drafted >= ?"
        )
        if engine_version:
            sql += " AND recommendation_engine_version = ?"
            params.append(engine_version)
        sql += " GROUP BY engine, status"
        rows = conn.execute(sql, params).fetchall()

    by_engine = {}
    for r in rows:
        by_engine.setdefault(r["engine"], {})[r["status"]] = r["n"]

    out = []
    for ev, counts in by_engine.items():
        active = counts.get("active", 0)
        expired = counts.get("expired_decayed", 0)
        reverted = counts.get("reverted", 0)
        pending = counts.get("pending_approval", 0) + counts.get("pending_evaluation", 0)
        total = active + expired + reverted + pending
        evaluated = total - pending
        unrevert = (active + expired) / evaluated if evaluated else 0.0
        revert = reverted / evaluated if evaluated else 0.0
        out.append(EngineQuality(
            engine_version=ev,
            n_recommendations=total,
            n_active=active,
            n_expired_decayed=expired,
            n_reverted=reverted,
            n_pending=pending,
            unrevert_rate=unrevert,
            revert_rate=revert,
            days_window=days,
        ))
    return out
```
